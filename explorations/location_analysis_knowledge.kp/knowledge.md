id_before = -1
start_loc = []
end_loc = []

for row in df.iterrows():
    current_id = row[1]['id']
    location = row[1]['lokasi_dominan']
    
    if (current_id != id_before):
        start_loc.append(location)
        end_loc.append(location)
    else :
        end_loc[-1] = location
    id_before = current_id
